1.1 排序

    快速排序：

int arr[N];

void quick_sort(int arr[],int l,int r){
    if(l>=r) return;
    //从左右端点两侧开始，因为下面do/while
    int i=l-1,j=r+1;
    //x取中间值可以防止边界问题
    int x = arr[l + r >> 1];
    while(i<j){
        do(i++);while(arr[i]<x);
        do(j--);while(arr[j]>x);
        //交换
        //这里的判断很重要，因为很可能会再一次循环中i>j
        //比如当j=i+2，i++之后arr[i]碰巧<x，会再次i++，然后j--就会导致i>j
        //i>j之后必然swap，所以没有这个if一定会出错。
        if (i < j) swap(arr[i], arr[j]);
    }
    quick_sort(arr,l,j);
    quick_sort(arr,j+1,r);
}

    归并排序

void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;
    //直接取中间值
    int mid = l + r >> 1;

    merge_sort(q, l, mid), merge_sort(q, mid + 1, r);

    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];
    //把剩余的加入数组
    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];
    //放回原数组
    for (i = l, j = 0; i <= r; i ++, j ++ ) q[i] = tmp[j];
}






1.2 二分

    最经典的二分就是二分查找，所有二分题都有类似的思路：想出一个check函数去检查当前二分点位是否满足某条性质（在二分查找中就是mid小于大于等于目标值），然后对不满足的情况进行if/else处理。虽然二分查找中依靠整个书里的单调性，但是二分本身是不需要单调性的：最重要的问题是边界（满足/不满足某个性质的边界）的判断。而二分的目的常常也是找到条件的边界。

1.2.1 整数二分

    首先假设红色区域代表符合某一性质的部分，绿色区域代表不符合，希望找到红色的边界。使用mid=(l+r+1)/2得到二分，然后使用check(mid)检查当前位置是否符合该性质：若符合 （落在了红色）那么更新下一次二分的范围为[mid,r]（需要包含mid，因为mid本身也可能是边界）；若不符合（落在了绿色），更新二分范围为[l,mid-1]（不包含mid因为mid必然不是符合某性质的边界）。

    另一种情况下，假设红色区域代表不符合某一性质的部分，绿色区域代表符合，希望找到绿色的边界。使用mid=(l+r)/2得到二分，然后使用check(mid)检查当前位置是否符合该性质：若符合 （落在了绿色）那么更新下一次二分的范围为[l,mid]（需要包含mid，因为mid本身也可能是边界）；若不符合（落在了红色），更新二分范围为[mid+1,r]（不包含mid因为mid必然不是符合某性质的边界）。

    可以看到如果我们想要找的边界是在形式上的左侧还是右侧会影响二分公式本身 以及更新后的二分范围。二分范围的变化很好理解，但问题是什么时候使用mid=(l+r+1)/2，什么时候使用mid=(l+r)/2。

    在C++中整形自动向下取整，假如在第一种情况中没有+1：若l=r-1，也就是说再来一轮操作就能确定边界在哪/是否存在了，mid=(l+r)/2会导致mid=l。若check(mid)为真，二分会进入[mid,r]，也就是[l,r]，即死循环。但是在第二种情况中不会出现这种情况：因为当l=r-1时，若check(mid)为真则变换范围为[l,l]，反之r,r，无论如何都结束循环且得到答案。

    这种情况是固定的，所以可以记住，只要范围变换的规律符合两种情况之一，那么对应的mid的计算公式就直接使用即可。

    例题：给定一个按照升序排列的长度为 n 的整数数组，以及 q 个查询。对于每个查询，返回一个元素 k 的起始位置和终止位置（位置从 0 开始计数）。如果数组中不存在该元素，则返回 -1 -1。

输入格式：

  第一行包含整数 n 和 q，表示数组长度和询问个数。第二行包含 n 个整数（均在 1∼10000 范围内），表示完整数组。接下来 q 行，每行包含一个整数 k，表示一个询问元素。

输出格式：

  共 q 行，每行包含两个整数，表示所求元素的起始位置和终止位置。如果数组中不存在该元素，则返回 -1 -1。

int n,m,p[N];

int main()
{
    cin>>n>>m;
    for (int i = 0; i < n; i ++ ) scanf("%d", &p[i]);
    
    while (m -- ){
        int x;
        scanf("%d", &x);
        int l=0,r=n-1;
        while(l<r){
            int mid=l+r>>1;
            if(p[mid]>=x) r=mid;
            else l=mid+1;
        }
        //没找到
        if(p[l]!=x) printf("-1 -1\n");
        else{
            printf("%d",l);
            l=0,r=n-1;
            while(l<r){
                int mid=l+r+1>>1;
                if(p[mid]<=x) l=mid;
                else r=mid-1;
            }
            printf(" %d\n",l);
        }
    }
}

首先画一个图，随便找两个点当x和mid，由于先输出起始点，就先思考“不符合的在左侧，符合的在右侧的情况”，然后得到范围变换的方式，便确定了mid的公式。之后输出终止点，也是一样。



1.2.2 浮点数二分

    相比于整数二分，浮点数二分不需要关心边界问题，因为不会取整。l与r可能永远不会相等（double），所以只要当l-r小于某个很小的值的时候，就说它俩相等

    例题：开平方根

int main(){
    double x;
    cin>>x;
    double l=0,r=x;
    double mid;
    //这里的差值最好比题目中精度高2
    while(r-l>1e-6){
        mid=(l+r)/2;
        if(mid*mid>x) r=mid;
        else l=mid;
    }
    cout<<mid;
}






1.3 高精度

    高精度分为高精度加法：大整数相加；高精度乘法：大整数与小整数相乘；高精度除法：大整数与小整数相除。



1.3.1 高精度加法

    本质上就是最原始的每一位相加。需要注意的是，为了进位的方便，需要把输入转换为逆序，即12345变成54321。只有加法不需要关心前导0.

//这里传入的是引用，底层中不会复制一份新的容器，效率更高
vector<int> add(vector<int> &a, vector<int> &b){
    //由于循环中的操作，a必须是更长的那个；等长时大小无关
    if(a.size()<b.size()) return add(b,a);
    vector<int> C;
    //t是一位上的答案，也是进位的指示
    int t=0;
    for (int i = 0; i < a.size(); i ++ ){
        t+=a[i];
        //需要在b的范围内操作
        if(i<b.size()) t+=b[i];
        C.push_back(t%10);
        //如果需要进位，t=1.
        t/=10;
    }
    //把最后可能有的进位补上
    if(t) C.push_back(t);
    
    return C;
}

int main()
{
    string a,b;
    cin>>a>>b;
    vector<int> A,B,C;
    //倒序排列
    //由于输入的是字符串，所以需要每个字符-‘0’得到正确的数值
    for (int i = a.size()-1; i >= 0; i -- ) A.push_back(a[i]-'0');
    for (int i = b.size()-1; i >= 0; i -- ) B.push_back(b[i]-'0');
    C=add(A,B);
    for (int i = C.size()-1; i >= 0; i -- ) cout<<C[i];
    return 0;
}


1.3.2 高精度减法

//创建一个比较函数
//减法中要求第一个参数大于等于后一个参数
bool cmp(vector<int> &a, vector<int> &b){
    //先对比长度
    if(a.size()!=b.size()) return a.size()>b.size();
    for (int i = 0; i < a.size(); i ++ ){
        //对比大小
        if(a[i]!=b[i]) a[i]>b[i];
    }
    return true;
}

vector<int> sub(vector<int> &a, vector<int> &b){
    vector<int> C;
    //指示计算结果和借位
    int t=0;
    //类似加法
    if(!cmp(a,b)) return sub(b,a);
    for (int i = 0; i < a.size(); i ++ ){
        //-t是因为可能的借位
        t=a[i]-t;
        //在b的范围内计算
        if(i<b.size()) t-=b[i];
        //计算到超出b的长度后，就是对a的复制了
        //(t+10)%10：假如t=-2，结果为(10-2)%10
        //假如t=5,结果为(10+5)%10；永远都是正数
        C.push_back((t+10)%10);
        //考虑是否借位
        if(t<0) t=1;
        else t=0;
    }
    //由于是逆序，可能出现321-021=300（也就是123-120）这种情况
    //需要把多余的0删掉
    while(C.size()>1&&C.back()==0) C.pop_back();
    return C;
}

int main()
{
    string a,b;
    cin>>a>>b;
    vector<int> A,B,C;
    for(int i=a.size()-1;i >= 0;i --) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i >= 0;i --) B.push_back(b[i]-'0');
    C=sub(A,B);
    //判断A是否小于B，若小于，首先要输出负号，也就是计算-（B-A）
    if(!cmp(A,B)) cout<<'-';
    for(int i=C.size()-1;i >= 0;i --) cout<<C[i];
    return 0;
    
}


1.3.3 高精度乘法

    本质上就是加法的推广。如111*19，实际上是前者的每一位和19相乘，而不是对应位置相乘。

vector<int> mul(vector<int> &a,int b){
    int t=0;
    vector<int> C;
    for (int i = 0; i < a.size() || t; i ++ ){
        //防止忽略t在最后的进位
        if(i<a.size()) t+=a[i]*b;
        //取个位作结果
        C.push_back(t%10);
        //向下取整，得到进位
        t/=10;
    }
    //如果不删前导0，可能出现111*0=000
    //这里是C.size()>1，因为可能结果为0
    while(C.size()>1&&C.back()==0) C.pop_back();
    
    return C;
}

int main()
{
    string a;
    int b;
    cin>>a>>b;
    vector<int> A,C;
    for (int i = a.size()-1; i >= 0; i -- ) A.push_back(a[i]-'0');
    C=mul(A,b);
    for (int i = C.size()-1; i >= 0; i -- ) cout<<C[i];
    return 0;
}


1.3.4 高精度除法

    除法不需要逆序运算，因为不同于前三个，它本身就是从最高位开始算的，符合输入的顺序。但是通常一道题里不会只有除法，为了统一规则，除法也逆序。除法运算中关键的一点是对竖式中每一层的余数的处理。比如100/3：

第一层商为0，余数为1：这个余数也就成了下一层的高位。

//由于需要输出余数，所以传入一个容器
vector<int> div(vector<int> &a, int b, int &r){
    vector<int> C;
    r=0;
    //计算过程
    //从高位开始
    for (int i = a.size()-1; i >= 0; i -- ){
        //实际上除法的计算是对余数不断的计算
        r=r*10+a[i];
        //得到商
        C.push_back(r/b);
        //得到下一层余数
        r%=b;
    }
    //由于上面是从高到低计算，push_back导致结果正序，需要再反过来（为了输出）
    reverse(C.begin(),C.end());
    //删除前导0
    while(C.size()>1&&C.back()==0) C.pop_back();
    
    return C;
}

int main()
{
    string a;
    int b,r;
    cin>>a>>b;
    vector<int> A,C;
    for (int i = a.size()-1; i >= 0; i -- ) A.push_back(a[i]-'0');
    C=div(A,b,r);
    for (int i = C.size()-1; i >= 0; i -- ) cout<<C[i];
    puts("");
    cout<<r;
    return 0;
}






1.4 前缀和与差分



1.4.1 一维前缀和

    前缀和的作用只有一个：用$\mathcal{O}(1)$ 求区间和，比如[l,r]的值为$S_r-S_{l-1}$ 。为了处理边界问题，前缀和数组第0位值为0，这样[1,n]的区间和就是[n]了。前缀和更多的是思想而不是算法。

//p：前缀和数组；a：原数组
int p[N],a[N];
int main()
{
    int n,m;
    scanf("%d%d", &n, &m);
    //输入数据
    for (int i = 1; i <= n; i ++ ) scanf("%d", &a[i]);
    //前缀和计算
    for (int i = 1; i <= n; i ++ ) p[i]=p[i-1]+a[i];
    //进行m个询问
    while (m -- ){
        int b,c;
        scanf("%d%d", &b, &c);
        //通过公式计算区间和
        printf("%d\n",p[c]-p[b-1]);
    }
    return 0;
    
}


1.4.2 二维（矩阵）前缀和

    类似一维前缀和，矩阵前缀和求的是$[x_1,y_1], [x_2,y_2]$ 两个坐标之间的和。

求前缀和矩阵某个元素的值的时候，计算的实际上是左元素+上元素-两者的重叠：也就是需要考虑容斥原理。同样地，找某个区间和的时候，如上图，也需要考虑容斥原理。 
int p[N][N],a[N][N];

int main()
{
    int n,m,q;
    scanf("%d%d%d", &n, &m, &q);
    //原数组赋值
    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= m; j ++ ){
            scanf("%d", &a[i][j]);
        }
        
    //求前缀和
    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= m; j ++ ){
            //容斥原理，并且加上当前位置的值
            p[i][j]=p[i-1][j]+p[i][j-1]-p[i-1][j-1]+a[i][j];
        }
    while(q--){
        int x1,y1,x2,y2;
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        //这里计算区间和，注意-1用来处理边际
        printf("%d\n",p[x2][y2]-p[x2][y1-1]-p[x1-1][y2]+p[x1-1][y1-1]);
    }
    
}


1.4.3 一维差分

    也就是前缀和的逆运算：给定一个数组，然后构造一个差分数组，使得前者是后者的前缀和。一维中，$b_n=a_n-a_{n-1}$，也就是相邻元素的差值。用处：给原数组中某个区间 $[l,r]$ 中的元素每个都增加 $c$，如果使用暴力解法，那就是 $\mathcal{O}(n)$，但是可以使用差分做到$\mathcal{O}(1)$：由于差分数组的前缀和是原数组，差分数组b[l]+=c会导致从l开始的通过b复原的a中元素全部加c，然后在b[r+1]-=c就可以让a[r]后面的元素保持不变。

#include <iostream>
#define N 100010
using namespace std;


int a[N],b[N];

void insert(int l, int r, int c){
    //在差分数组l的位置增加c，后面所有也增加c
    b[l]+=c;
    //抵消上面的操作
    b[r+1]-=c;
}

int main(){
    int n,m;
    cin>>n>>m;
    //原数组
    for (int i = 1; i <= n; i ++ ) scanf("%d", &a[i]);
    //构建差分数组
    for (int i = 1; i <= n; i ++ ) insert(i,i,a[i]);
    //执行操作
    while (m -- ){
        int l,r,c;
        scanf("%d%d%d", &l, &r,&c);
        insert(l,r,c);
    }
    //还原前缀和数组
    for (int i = 1; i <= n; i ++ ) b[i]+=b[i-1];
    for (int i = 1; i <= n; i ++ ) printf("%d ",b[i]);
    
    return 0;
}

上面的代码构建差分数组的方式实际上就是通过insert函数把左右端点都看作l，只对这一个位置进行操作。但是由于数组默认值全部是0，就可以构建出差分数组（之后一个位置的值是0-a[l]+a[l+1]）。这个构建的过程实际上可以把a，b两个数组都看作最开始全部值为0的两个数组。然后对于每个a上的位置的赋值可以理解为对该位置的插入，即增加c。



1.4.4 矩阵差分

    类似前缀和矩阵，差分矩阵的目的是给一个子矩阵内的所有值+c

如上图所示，对于$[x_1,y1]$ 的增加会导致整个包含的部分都改变，所以需要处理：减小$[x_2+1,y1],[x_1,y_2+1]$，并且还原增加重叠的部分$[x_2+1,y_2+1]$。对待差分矩阵的构建遵循和一维一样的思路，即假设原数组全部为0，然后执行了`m*n`次插入操作。 
#include <iostream>
#define N 1010
using namespace std;


int n,m,q;
int a[N][N],b[N][N];


void insert(int x1, int y1, int x2, int y2, int c){
    //增加操作
    b[x1][y1]+=c;
    //抵消操作
    b[x1][y2+1]-=c;
    b[x2+1][y1]-=c;
    b[x2+1][y2+1]+=c;
}


int main()
{
    scanf("%d%d%d", &n, &m, &q);

    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= m; j ++ )
            scanf("%d", &a[i][j]);
    //用同样的逻辑构建差分数组
    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= m; j ++ )
            insert(i, j, i, j, a[i][j]);

    while (q -- )
    {
        int x1, y1, x2, y2, c;
        scanf("%d%d%d%d%d", &x1, &y1,&x2, &y2, &c);
        insert(x1, y1, x2, y2, c);
    }

    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= m; j ++ )
            //还原数组的时候也要关注容斥原理
            b[i][j] += b[i - 1][j] + b[i][j - 1] - b[i - 1][j - 1];

    for (int i = 1; i <= n; i ++ )
    {
        for (int j = 1; j <= m; j ++ ) printf("%d ", b[i][j]);
        puts("");
    }

    return 0;
}






1.5 双指针

    其中一种情况就像是在归并排序中合并序列是使用的两个指针分别指向两个数组，但是更常见的一种情况是两个指针指向一个序列：

for(int i=0, j=0;i <= n; i++){
    //check检查是否满足某一性质
    while(j < i && check(i,j)) j++;

    //下面是每道题目的具体逻辑
}

这就是双指针常见的模板。双指针最核心的优势是可以优化$\mathcal{O}(n^2)$ 的算法为$\mathcal{O}(n)$，因为两个指针遍历元素所移动的次数不可能超过 $2n$。

    例子：给定一个长度为 n 的整数序列，请找出最长的不包含重复的数的连续区间，输出它的长度。

int a[N],b[N];

int main()
{
    int n,ans=0;
    cin>>n;
    for (int i = 0; i < n; i ++ ) scanf("%d", &a[i]);
    
    for (int i = 0,j=0; i < n; i ++ ){
        b[a[i]]++;
        //如果i上的值的个数>1,证明有数字重复并且最新的i对应的a中的数字造成的
        //b[a[j++]]--逐渐向前走，所以需要清除对于跳过了的数字的计数
        //当循环条件后半部分不成立的时候，证明区间内无重复
        while( j<i && b[a[i]]>1) b[a[j++]]--;
        ans=max(ans,i-j+1);
    }
    cout<<ans;
    return 0;
}

使用数组b来装数组a中每个出现的元素的次数，后者中的值为前者中的下标。

注： 双指针算法的题目就先想一下暴力算法，然后思考是否存在任何单调关系（j不会往回走）。

1.7 离散化

    这里的离散化特指整数的离散化。在解题过程中通常会把输入数据上下限作为数组长度来储存这些数据，但是当数据范围过大时会造成数组过大，离散化用来把数组长度压缩为输入数据的个数：本质上实现了一种 数值-坐标 的映射

例题：

    假定有一个无限长的数轴，数轴上每个坐标上的数都是 0。现在，我们首先进行 n 次操作，每次操作将某一位置 x 上的数加 c。接下来，进行 m 次询问，每个询问包含两个整数 l 和 r，你需要求出在区间 [l,r] 之间的所有数的和。

输入格式：

  第一行包含两个整数 n 和 m。接下来 n 行，每行包含两个整数 x 和 c。再接下来 m 行，每行包含两个整数 l 和 r。

输出格式：

  共 m 行，每行输出一个询问中所求的区间内数字和。

数据范围 $−10^9\leq x\leq 10^9$,

$1\leq n,m\leq 10^5$,

$−10^9\leq l\leq r\leq 10^9$,

$−10000\leq c\leq 10000$

//由于n,m 2个变量的大小都是10000，那么最多总共涉及的坐标数量为2m+n，即300000
#define N 300030

//dis是离散化后的坐标数组，不通过映射，而是直接排序消重得到的
//它包括题目中“添加的目标index”以及“查询的index”
//为的是让每个index都能独立的占有一个离散化之后的位置，未来通过find才能找到目标位置
vector<int> dis;
//用pair保存操作
vector<pair<int,int>> add,query;
//inp：保存添加后的数组；s：前缀和数组
int inp[N],s[N];
using namespace std;

//离散化函数
//思想：找到所有输入中比目标坐标x大的最小的值；使用二分
//这样做是保证区间正确：如果不是最小的，会导致区间会多出计算的元素
int find(int x){
    int l=0, r=dis.size()-1;
    int mid;
    while(l<r){
        mid=l+r>>1;
        if(dis[mid]>=x) r=mid;
        else l=mid+1;
    }
    //+1为了从1开始计算区间和
    return r+1;
}

int main()
{
    int n,m;
    int a,b;
    cin>>n>>m;
    //保存添加操作，把目标坐标全放入dis
    for (int i = 0; i < n; i ++ ){
        scanf("%d%d", &a, &b);
        add.push_back({a,b});
        dis.push_back(a);
    }
    //同理
    for (int i = 0; i < m; i ++ ){
        scanf("%d%d", &a, &b);
        query.push_back({a,b});
        dis.push_back(a);
        dis.push_back(b);
    }
    //映射需要借助单调性，所以需要排序
    sort(dis.begin(),dis.end());
    //unique把不重复的元素放在前面，不重复的元素个数>=原个数
    //不重复部分之后的部分是原vector同一位置上之前的值
    //erase把特定范围删掉
    //必须这样做，因为映射需要元素唯一，不然find可能返回不同的r
    dis.erase(unique(dis.begin(),dis.end()),dis.end());
    
    for (auto item : add)
    {
        //通过find找到对应的映射
        int x = find(item.first);
        inp[x] += item.second;
    }
    //前缀和
    for (int i = 1; i <= dis.size(); i ++ ) s[i]=s[i-1]+inp[i];
    //查找
    for (auto item : query)
    {
        int l = find(item.first), r = find(item.second);
        cout << s[r] - s[l - 1] << endl;
    }
}






1.8 区间合并

例题：

    给定 n 个区间 $[l_i,r_i]$，要求合并所有有交集的区间。注意如果在端点处相交，也算有交集。输出合并完成后的区间个数。例如：[1,3] 和 [2,6] 可以合并为一个区间 [1,6]。

输入格式:

  第一行包含整数 n。接下来 n 行，每行包含两个整数 l 和 r。

输出格式:

  共一行，包含一个整数，表示合并区间完成后的区间个数。


#define N 1000010
typedef pair<int, int> PII;

vector<PII> segs;

void merge(vector<PII> &segs){
    vector<PII> res;
    //使用两个极值作为起始点
    int st=-2e9,ed=-2e9;
    //根据左端点排序，类似贪心
    sort(segs.begin(), segs.end());
    for(auto seg:segs){
        //对于不重合区间的处理
        if(ed<seg.first){
            //需要检查不是默认极值
            if(st!=-2e9) res.push_back({st,ed});
            st=seg.first,ed=seg.second;
        }
        //重合的处理：更新右端点
        else ed=max(ed,seg.second);
    }
    //最后一个区间不会被上面的循环添加
     if (st != -2e9) res.push_back({st,ed});
    segs=res;
}
int main()
{
    int n;
    cin>>n;
    
    while (n -- ){
        int a,b;
        scanf("%d%d", &a, &b);
        segs.push_back({a,b});
    }
    merge(segs);
    
    cout<<segs.size();
    
    
}
