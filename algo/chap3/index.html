3.1 深度优先

    DFS每次会搜索到叶节点再回溯。DFS使用stack，空间复杂度为$\mathcal{O}(h)$，也就是说大小和高度成正比。不具有最短性。DFS中最重要的两个概念：回溯，剪枝。

    例题：排列数字：给定一个整数 n，将数字 1∼n 排成一排，将会有很多种排列方法。现在，请你按照字典序将所有的排列方法输出。

  输入格式：

    共一行，包含一个整数 n。

  输出格式：     按字典序输出所有排列方案，每个方案占一行。

  数据范围：$1\leq n \leq 7$.

int path[N];
int n;

void DFS(int u, int state){
    //如果遍历到最后一个元素了，说明可以位数足够输出了
    if(u==n){
        for (int i = 0; i < n; i ++ )printf("%d ",path[i]);
        puts("");
    }
    //遍历每一个数字
    for (int i = 0; i < n; i ++ ){
        //检查state第i个bit的值，如果是0证明对应的数字（i+1）还没有被使用
        if(!(state>>i&1)){
            //+1得到对应的数字
            path[u]=i+1;
            //递归path下一位，设置state对应位置为“被使用”
            //这里不用担心在下次递归的时候state对应位置会出问题，因为由于传入形参作为局部变量存在，不会对着一层的state造成影响
            DFS(u+1,state+(1<<i));
            }
    }
}

int main(){
    cin>>n;
    //从根节点开始，对应着遍历的最左侧的数字
    //第二个0代表state，使用压位，每一位代表一个元素的状态
    DFS(0,0);
}

    例题：n−皇后问题是指将 n 个皇后放在 n×n 的国际象棋棋盘上，使得皇后不能相互攻击到，即任意两个皇后都不能处于同一行、同一列或同一斜线上。

现在给定整数 n，请你输出所有的满足条件的棋子摆法。

  输入格式:

    共一行，包含整数 n。

  输出格式:

    每个解决方案占 n 行，每行输出一个长度为 n 的字符串，用来表示完整的棋盘状态。其中 . 表示某一个位置的方格状态为空，Q 表示某一个位置的方格上摆着皇后。每个方案输出完成后，输出一个空行。注意：行末不能有多余空格。输出方案的顺序任意，只要不重复且没有遗漏即可。

  数据范围:

    $1\leq n \leq 9$.

  思路：n-皇后问题的解决思路和前一道题类似。本质上还是使用DFS的暴力搜索，遍历每一行每一列的可能性。可以通过剪枝来进行优化，也就是说当某一方案在遍历的过程中发现下一个元素在添加之后会导致它不可能行，那么就不再往这个元素之后遍历了。复杂度为$n*n!$。

char g[N][N];
//用来检查本列/反对角线/正对角线是否有Q
bool col[N],dg[N],udg[N];
int n;

void DFS(int r){
    //输出
    if(r==n) {
        for (int i = 0; i < n; i ++ ) puts(g[i]);
        puts("");
        return;
    }
    //遍历本行每一列
    for (int i = 0; i < n; i ++ ){
        //如果本列和本两个对角线都没有Q，进入递归
        //相反则剪枝
        //这里的r+i把横纵坐标映射到了一条对角线
        //i-r+n也是一样的，但是由于i-r可能是负数，所以使用+n修正
        if(!col[i]&&!dg[n-r+i]&&!udg[r+i]){
            g[r][i]='Q';
            //标记当前位置为存在Q
            col[i]=dg[n-r+i]=udg[r+i]=1;
            //递归下一行
            DFS(r+1);
            //还原：每个递归都适用全局的三个数组，所以回溯之后得保证数据不变
            col[i]=dg[n-r+i]=udg[r+i]=0;
            //这里也要还原
            g[r][i] = '.';
        }
    }
    
    
}

int main()
{
    cin>>n;
    //初始化
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < n; j ++ )
            g[i][j]='.';
    DFS(0);
    return 0;
}

    总结：DFS是一种思想，其中最重要的是对遍历“顺序”的考量；没有确切的模板。







3.2 广度优先

    BFS每次搜索一层。BFS使用queue，空间复杂度为$\mathcal{O}(2^h)$，因为每次搜索会把一整层的节点全部储存下来。具有最短路的特点：第一次搜索到某元素的时候，一定会通过最短路找到该元素的（默认edge权重都为1）。BFS也是一种思想，可以用来解决最短路问题（边权重都是1）。有常用框架。

    例题：给定一个 n×m 的二维整数数组，用来表示一个迷宫，数组中只包含 0 或 1，其中 0 表示可以走的路，1 表示不可通过的墙壁。最初，有一个人位于左上角 (1,1) 处，已知该人每次可以向上、下、左、右任意一个方向移动一个位置。请问，该人从左上角移动至右下角 (n,m) 处，至少需要移动多少次。数据保证 (1,1) 处和 (n,m) 处的数字为 0，且一定至少存在一条通路。

  输入格式：

    第一行包含两个整数 n 和 m。接下来 n 行，每行包含 m 个整数（0 或 1），表示完整的二维数组迷宫。

  输出格式：

    输出一个整数，表示从左上角移动至右下角的最少移动次数。

  数据范围：

    $1\leq n,m\leq 100$

#define N 110

//g是原迷宫，d是每个点被遍历时距离原点的距离
int g[N][N],d[N][N];
int m,n;

int BFS(){
    queue<PII> q;
    //初始化距离为-1,memset在cstring
    memset(d,-1,sizeof d);
    d[0][0]=0;
    //先把第一个点入queue
    q.push({0, 0});
    //这里设置上下左右移动
    int dx[4]={-1,0,1,0}, dy[4]={0,1,0,-1};
    while(q.size()){
        PII t=q.front();
        q.pop();
        //遍历四个方向
        for (int i = 0; i < 4; i ++ ){
            int x=t.first+dx[i], y=t.second+dy[i];
            //即不能超出边界，是第一次被遍历的可以走过去的位置
            if(x>=0&&y>=0&&x<n&&y<m&&d[x][y]==-1&&g[x][y] == 0){
                q.push({x,y});
                //更新距离
                d[x][y]=d[t.first][t.second]+1;
            }
        }
    }
    return d[n-1][m-1];
}

int main()
{
    cin>>n>>m;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            scanf("%d", &g[i][j]);
    cout<<BFS();
}

如果想要回溯路径，可以新建一个二维数组记录每个点是从哪个点来的。







3.3 图

    树是无环连通图，无向图是双向图，所以只讨论图。图有两种储存方式：

  1. 邻接矩阵：适合储存稠密图，g[a][b]代表a到b是否连接；

  2. 邻接表：长得和拉链法的哈希表一模一样：是一个单链表，元素数量等于图中节点数量，每个元素也是一个单链表，表示该节点与表中的节点相连；其元素插入和拉链法也一样。

这里通过邻接表实现宽度优先遍历：

const int N = 100010, M = N * 2;
//h：头节点的链表；e：每个元素都是一个图中的节点，可重复
//ne：下一个节点的idx；
int h[N], e[M], ne[M], idx;
//st:查看当前点是否被遍历过
//st[j]，j表示某个节点而不是某个节点的idx
//因为st表示的是这个节点的状态，一个节点可以有多个idx
bool st[N];

//添加连接关系的函数：
//a是目标节点，b是a所指向的一个节点
//把b插入a的链表
void add(int a, int b){
    //插在表头
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

//图的深度遍历
//u是起始节点
void dfs(int u){
    st[u] = true;
    //遍历拉链
    for(int i = h[u]; i != -1; i = ne[i]){
        //提取出拉链中元素对应的节点
        int j = e[i];
        //查看状态，递归
        if (!st[j]) dfs(j);
    }
}

int main(){
    memset(h,-1,sizeof h);
}

    例题：给定一颗树，树中包含 n 个结点（编号 1∼n）和 n−1 条无向边。请你找到树的重心，并输出将重心删除后，剩余各个连通块中点数的最大值。重心定义：重心是指树中的一个结点，如果将这个点删除后，剩余各个连通块中点数的最大值最小，那么这个节点被称为树的重心。

输入格式
第一行包含整数 n，表示树的结点数。

接下来 n−1 行，每行包含两个整数 a 和 b，表示点 a 和点 b 之间存在一条边。

输出格式
输出一个整数 m，表示将重心删除后，剩余各个连通块中点数的最大值。

数据范围
1≤n≤105

int e[2*N],ne[2*N],idx,n,h[N],ans=N;
bool st[N];

void add(int a, int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
//主要思路就是通过dfs找到输入节点为跟节点的树里的元素个数
int dfs(int u){
    st[u]=true;
    int sum=0,size=0;
    //遍历邻接表
    for (int i = h[u]; i != -1; i = ne[i] ){
        if(st[e[i]]) continue;
        //s是子树的元素数量
        int s = dfs(e[i]);
        //这里判断删除输入节点后产生的所有子树的最大元素数量
        size=max(s,size);
        //这里计算这个树的元素数量
        sum+=s;
    }
    //n-sum-1求出删除该节点后，其父节点所在的树的元素数量
    size=max(size,n-sum-1);
    //求删除该节点后的最大树与删除其他节点后的最大树的最小树
    ans=min(size,ans);
    //返回包含输入节点的子树元素数量
    return sum+1;
    
}

int main()
{
    memset(h, -1, sizeof h);
    cin>>n;
    //无向图即双向
    for (int i = 0; i < n-1; i ++ ){
        int a,b;
        scanf("%d%d", &a, &b);
        add(a,b);
        add(b,a);
    }
    dfs(1);
    cout<<ans;
}

    广度优先遍历：例题：给定一个 n 个点 m 条边的有向图，图中可能存在重边和自环。所有边的长度都是 1，点的编号为 1∼n。请你求出 1 号点到 n 号点的最短距离，如果从 1 号点无法走到 n 号点，输出 −1。

输入格式
第一行包含两个整数 n 和 m。

接下来 m 行，每行包含两个整数 a 和 b，表示存在一条从 a 走到 b 的长度为 1 的边。

输出格式
输出一个整数，表示 1 号点到 n 号点的最短距离。

数据范围
1≤n,m≤105

int ne[N],e[N],d[N],q[N],n,m,idx,h[N];

void add(int a, int b){
    e[idx]=b, ne[idx]=h[a], h[a]=idx++;
}

int bfs(){
    //tt==hh说没有一个元素
    int tt=0,hh=0;
    q[0]=1;
    memset(d, -1, sizeof d);
    d[1]=0;
    //当queue不为空
    while(hh<=tt){
        //出队
        int t=q[hh++];
        //遍历当前节点邻接点
        for (int i = h[t]; i != -1; i = ne[i] ){
            int j = e[i];
            if(d[j]==-1){
                //计算下一次所有节点距离
                d[j]=d[t]+1;
                q[++tt]=j;
            }
        }
    }
    return d[n];
}

int main()
{
    cin>>n>>m;
    memset(h, -1, sizeof h);
    for (int i = 0; i < m; i ++ ){
        int a,b;
        scanf("%d%d", &a, &b);
        add(a, b);
    }
    cout<<bfs();
    
}






3.3 拓扑排序

    拓扑排序指对顶点排序之后，所有边都是从左往右指。 DAG：有向无环图一定有拓扑排序。其中有两个概念：入度：指向该顶点的边的数量，出度：该顶点指向其他定点的边。所有入度为0的节点可以作为排序起始点。一个DAG必有一个入度为0的顶点。找到一个拓扑排序：

//d[N]代表顶点的入度大小
int q[N],e[N],ne[N],h[N],n,m,idx,d[N];

void add(int a, int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
bool toposort(){
    int hh=0,tt=-1;
    //找到入度为0的顶点，入队
    for (int i = 1; i <= n; i ++ ){
        if(!d[i]) q[++tt]=i; 
    }
    
    while(hh<=tt){
        int t=q[hh++];
        for (int i = h[t]; i != -1; i = ne[i] ){
            int j=e[i];
            //相当于剪掉两个顶点之间的边，后者入度--
            //如果入度为0，证明可以入队了
            if(-- d[j] == 0) q[++tt]=j;
        }
    }
    //如果不为真，证明环存在
    return tt==n-1;
}

int main()
{
    cin>>n>>m;
    memset(h, -1, sizeof h);
    for (int i = 0; i < m; i ++ ){
        int a,b;
        scanf("%d%d", &a, &b);
        add(a, b);
        //入度增加
        d[b]++;
    }
    if(!toposort()) puts("-1");
    else {
        for (int i = 0; i < n; i ++ ) printf("%d ",q[i]);
    }
}

在剪掉一个顶点的入度之后，只有入度==0才能进入队列，因为如果不是这样，可能造成后面入队的某个顶点指向它。如果有环，会导致队列为空后仍然有没有入队的顶点。







3.4 最短路

    不同的最短路算法对应不同的类型。通常分为两大类：

  1. 单源最短路：从一个点到其他所有点的最短路；

    又分为：

      所有边权重为正：

        朴素Dijkstra $O(|V|^2)$，复杂度与边数无关，适合稠密图;

        堆优化Dijkstra $O(|E|\log |V|)$

      存在负权变：

        Bellman-Ford $O(|V||E|)$

        SPFA 一般$O(|E|)$，最坏$O(|V||E|)$，是BF的优化

  1. 多源汇最短路：多个起点终点的最短路：Floyd算法 $O(|V|^3)$

注：$|E|\approx |V|^2$ 以上被称为稠密图，$|E|\approx |V|$ 被称为稀疏图；源：起点；汇：终点。

    最短路问题的难点在于把题目抽象成图，而非正确性计算。最短路算法可以直接解决无向图的问题，因为无向图实际上就是特殊的有向图。



3.4.1 朴素Dijkstra

    概念：**松弛操作（Relaxation）**是指对于一条从顶点u指向顶点v的边 u-->v来说，如果满足 d[u]+w(u,v)<d[v],就更新 d[v],使得 d[v]=d[u]+w(u,v)；这就是对 边uv的一次放松操作；Dijkstra算法和Bellman-Ford都使用了这一思想。

    中心思想：从起始点开始遍历它邻接的点，更新这些点到它的距离，选出全局最短的，这个最短的点可以被确定为找到了起点到它的最短路。再从后者做一样的事情，更新其邻接点到源点距离为原距离和当前点距原点距离+当前点与目标点距离中的最小值，然后再找全局最短的（不是找和这个顶点相连的最短的）。朴素Dijkstra应用于稠密图。

    例题：给定一个 n 个点 m 条边的有向图，图中可能存在重边和自环，所有边权均为正值。请你求出 1 号点到 n 号点的最短距离，如果无法从 1 号点走到 n 号点，则输出 −1。

输入格式
第一行包含整数 n 和 m。

接下来 m 行每行包含三个整数 x,y,z，表示存在一条从点 x 到点 y 的有向边，边长为 z。

输出格式
输出一个整数，表示 1 号点到 n 号点的最短距离。

如果路径不存在，则输出 −1。

数据范围
1≤n≤500,
1≤m≤10^5,
图中涉及边长均不超过10000。

可以看出这是一个稠密图。注：本题可能出现自环（指向自己的变）和重边（两个不同的{a,b}）。


//分别是图，对应顶点状态：是否加入了已被判定的点，距离数组
int g[N][N], st[N], dist[N];
int n,m;

int dj(){
    memset(dist,0x3f,sizeof dist);
    //顶点从1开始
    dist[1]=0;
    //n-1是因为最后一个点不需要遍历，并不是因为第一个节点被遍历
    //实际上第一个节点并没有被遍历
    //对于第一个节点的检索初始化了其四周的节点的距离
    for (int i = 0; i < n-1; i ++ ){
        //t为最短节点
        int t=-1;
        //找出最近的节点
        //这里是朴素版DJ可能导致过慢的地方，因为遍历了所有顶点
        for (int j = 1; j <= n; j ++ ){
            if(!st[j]&&(t==-1||dist[t]>dist[j])) t=j;
        }
        //把当前节点考虑入“已找到最短距的节点”中
        st[t]=1;
        //更新距离，本质为贪心
        for (int j = 1; j <= n; j ++ ){
            dist[j]=min(dist[j],dist[t]+g[t][j]);
        }
    }
    //未连接
    if(dist[n]==0x3f3f3f3f) return -1;
    return dist[n];
    
}

int main()
{
    cin>>n>>m;
    //初始化图，正无穷代表不连接
    memset(g, 0x3f, sizeof g);
    while (m -- ){
        int a,b,c;
        scanf("%d%d%d", &a, &b,&c);
        //防止重边
        g[a][b]=min(c,g[a][b]);
    }
    cout<<dj();
    
}

在上述算法中，更新t到其他点的距离总共$|E|$次，因为每条边只会被遍历一次；把t放入st数组总共$|V|$次，因为每个顶点被放入一次；最耗时间的步骤是查找全局最短节点，为$n^2$。由于“最近的节点”实际上是找一堆数字中最小的数字，可以用堆优化。



3.4.2 堆优化Dijkstra

    堆优化的Dijkstra用来处理稀疏图，这样更新t到其他点的距离花费$O(|E|\log |V|)$ （稠密图的|E|太大，所以用朴素），找最近节点变成$O(1)$。在堆优化中不需要关心重边问题，因为自动会选择最小的。

int h[N],e[N],ne[N],w[N],st[N],dist[N],idx,n,m;

void add(int a, int b,int c){
    w[idx]=c,e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}

int dj(){
    memset(dist, 0x3f, sizeof dist);
    //构建小根堆
    //每个元素为pair：距离，顶点编号；距原点距离作为大小判断
    priority_queue<PII, vector<PII>,greater<PII>> heap;
    dist[1]=0;
    heap.push({0,1});
    
    while(heap.size()){
        //获得全局最小值
        PII t=heap.top();
        heap.pop();
        //这里只需要顶点编号，不需要对应距离
        int ver=t.second;

        //一个顶点可能由于重边存在被重复录入在一个h[a]的链表上
        //造成两次入堆
        //这里再次访问一个被访问的节点，跳过
        if(st[ver]) continue;
        st[ver]=true;

        //更新邻接节点距离
        for (int i = h[ver]; i != -1; i = ne[i] ){
            int j=e[i];
            if(dist[j]>dist[ver]+w[i]) {
                dist[j]=dist[ver]+w[i];
                //入heap，直接得到全局最小
                heap.push({dist[j],j});
            }
        }
    }
    if (dist[n] == 0x3f3f3f3f) return -1;
    return dist[n];
    
}
int main()
{
    cin>>n>>m;
    memset(h, -1, sizeof h);
    for (int i = 0; i < m; i ++ ){
        int a,b,c;
        scanf("%d%d%d", &a, &b,&c);
        add(a, b, c);
    }
    cout<<dj();
}


3.4.3 Bellman-Ford 算法

    概念：三角不等式：在执行完BF之后，所有点都满足dist[b] <= dist[a] + w (a->b)。

    BF可以处理存在负权的图，但是存在负权回路的图不一定存在最短路：也就是说如果一个环的总长度是一个负数，那么可能会死循环这个圈，让总距离越来越少：

在进入bcd后会一直循环，每次循环总长度-1.BF算法可以用来检测是否存在这样的环：BF中外层循环的意义是保证在第k次循环开始时，找到了可以通过不超过k条边达到所有节点的最短路。而在循环到$|V|$ 时，若第$|V|$ 次还有更新，证明更新了$|V| + 1$ 个节点，一定有两个节点是一个（鸽子笼原理）即环存在，并且是负环（不然最短路算法不会走这条环）。但是找负环一般用SPFA算法。有的时候负环无影响：当负环不出现在目标源汇的路径上时。

    SPFA算法每一方面都优于BF，但是只有后者可以解决“指使用k条边找到最短路”这样的题。

    时间复杂度：$O(mn)$。

    例题：给定一个 n 个点 m 条边的有向图，图中可能存在重边和自环， 边权可能为负数。请你求出从 1 号点到 n 号点的最多经过 k 条边的最短距离，如果无法从 1 号点走到 n 号点，输出 impossible。注意：图中可能 存在负权回路 。

输入格式
第一行包含三个整数 n,m,k。

接下来 m 行，每行包含三个整数 x,y,z，表示存在一条从点 x 到点 y 的有向边，边长为 z。

点的编号为 1∼n。

输出格式
输出一个整数，表示从 1 号点到 n 号点的最多经过 k 条边的最短距离。

如果不存在满足条件的路径，则输出 impossible。

数据范围
1≤n,k≤500,
1≤m≤10000,
1≤x,y≤n，
任意边长的绝对值不超过 10000。

int n,m,k;
int dist[N],last[N];

//这里通过结构体存储更好，因为之后遍历是对于
//每个边的遍历，并不以节点为出发点，所以不需要
//像dj里需要知道邻接
struct Edge{
    int a,b,c;
}edges[M];

void bf(){
    //初始化距离为最大值
    memset(dist, 0x3f, sizeof dist);
    dist[1]=0;
    
    //每次循环都更新出一条“通过i条边可以达到的最短路径”
    //当i==n-1说明每一个顶点的最短路都被找到了
    for (int i = 0; i < k; i ++ ){
        //防止串联更新
        memcpy(last,dist,sizeof dist);
        //遍历每一条边，并更新“在多考虑一条边后，所有节点最短路”
        for (int j = 0; j < m; j ++ ){
            auto e = edges[j];
            dist[e.b]=min(dist[e.b],last[e.a] + e.c);
        }
    }
}

int main()
{
    cin>>n>>m>>k;
    for (int i = 0; i < m; i ++ ){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        edges[i].a=a,edges[i].b=b,edges[i].c=c;
    }
    bf();
    //这里使用 0x3f3f3f3f / 2 是因为disf[n]可能无法通过
    //k条边抵达，但是由于每次遍历所有边，dist[n]可能由于
    //n-1和n的边是负权，min(dist[e.b],last[e.a] + e.c)会导致
    //dist[n]=0x3f3f3f3f+e.c，变成一个更小的值
    if(dist[n] > 0x3f3f3f3f / 2) cout<<"impossible";
    else cout<<dist[n];
}

串联： 由于这个算法的特性决定，每次更新得到的必然是在多考虑 1 条边之后能得到的全局的最短路。而串联指的是一次更新之后考虑了不止一条边：由于使用了松弛，某节点的当前最短路依赖于其所有入度的节点的最短路；假如在代码中使用dist[e.b]=min(dist[e.b],dist[e.a] + e.c);，我们无法保证dist[e.a]是否也在本次循环中被更新，如果被更新了，并且dist[e.b] > dist[e.a] + e.c，那么会造成当前节点在事实上“即考虑了一条从某个节点指向a的边，也考虑了a->b”，共两条边。而使用dist[e.b]=min(dist[e.b],last[e.a] + e.c);，可以保证a在dist更新后不影响对b的判定，因为后者使用last数组，保存着上一次循环中的dist的值。

负环：出现负环后，dist[n]的值可能无意义：通常外部循环$|V|$ 次，会导致负环上的节点的dist减小$|V|$ 次，如果这个负环在源汇路径上，会导致dist[n]减小$x$次，$x<|V|$。



3.4.4 SPFA算法

    SPFA(Shortest Path Faster Algorithm) 是限制最少的最短路算法（正权负权都可以），只要没有负环就能用；是BF的优化：可以看到dist[e.b]=min(dist[e.b],last[e.a] + e.c);对每一条边进行更新，而其中很多边的大小是不会改变的，因为改变的唯一原因是dist[e.a]变小了，下一次才会更新dist[e.b]；这是浪费资源的，也是SPFA优化的地方。优化使用BFS。 在实践中，Dijkstra可能没有SPFA快。

//很大一部分写法和Dijkstra类似
int e[N],w[N],ne[N],h[N],st[N],n,m,idx,dist[N];

void add(int a,int b, int c){
    w[idx]=c,e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
void spfa(){
    memset(dist, 0x3f, sizeof dist);
    dist[1]=0;
    queue<int> q;
    //把源点放入queue
    q.push(1);
    st[1]=true;
    
    while(q.size()){
        int t=q.front();
        q.pop();
        //SPFA不存在“记录已经知道最短路的节点”这一概念
        //BF也没有
        st[t] = false;
        //遍历当前节点所有出边节点
        for (int i = h[t]; i != -1; i = ne[i] ){
            int j=e[i];
            //尝试更新距离
            //没有这个判断会导致死循环
            if(dist[j]>dist[t]+w[i]){
                dist[j]=dist[t]+w[i];
                //如果没有入q就入，不能重复入q
                if(!st[j]){
                    q.push(j);
                    st[j]=true;
                }
            }
        }
    }
    
}

int main()
{
    memset(h, -1, sizeof h);
    cin>>n>>m;
    while (m -- ){
        int a,b,c;
        scanf("%d%d%d", &a, &b, &c);
        add(a, b, c);
    }
    spfa();
    if(dist[n]==0x3f3f3f3f) puts("impossible");
    else cout<<dist[n];
}

主要思想就是通过一个queue去储存所有被修改最短路的节点，因为这些节点的所有出边指向的节点的最短路都可能可以被更新。这个算法不需要考虑串联，因为它没有这一概念（而不是自动防止了串联），所以也就不能解决想上一道题那种题目。

    判断负环：代码基本相同

//cnt保存当前顶点到起始点之间边的数量
int cnt[N];

bool spfa(){
    queue<int> q;
    //把所有顶点放入queue
    for (int i = 1; i <= n; i ++ ) {
        q.push(i);
        st[i]=1;
    }
    
    while(q.size()){
        int t=q.front();
        q.pop();

        st[t] = false;
        for (int i = h[t]; i != -1; i = ne[i] ){
            int j=e[i];
            if(dist[j]>dist[t]+w[i]){
                dist[j]=dist[t]+w[i];
                //这里是最重要的步骤
                //当j被抵达的时候，如果使用的边的数量=n，
                //证明从源点到j经过了n+1的顶点，证明其中有重复
                cnt[j]=cnt[t]+1;
                if(cnt[j]>=n) return true;
                if(!st[j]){
                    q.push(j);
                    st[j]=true;
                }
            }
        }
    }
    return false;
    
}

注：需要把所有顶点放入queue，因为假如只把1放入的话，会导致只能找到1可以走到的负环，而不一定所有负环都能从1走到；如果题目要求只找负环的话，就不需要初始化dist了，因为负环一定会导致if(dist[j]>dist[t]+w[i])为真；如果没有return语句，会死循环。



3.4.5 Floyd算法

    本质上是一种动态规划（背包问题）。对所有点考虑“如果经过某点是否可以得到达到另一个点更近的距离”：也就是比较dist[a,b]<=dist[a,c]+dist[c,b]。可以找到任意一点到任意一点的最短路（多源汇最短路）。

int n,m,k;
//储存着图
int d[N][N];

//三重循环
void floyd(){
    //这里最外层循环就是两点之间的中间点
    //用来检查是否dist[a,b]<=dist[a,c]+dist[c,b]
    for (int k = 1; k <= n; k ++ )
        // i和j就是上面的a和b，k就是c
        for (int i = 1; i <= n; i ++ )
            for (int j = 1; j <= n; j ++ )
                d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
}

int main()
{
    cin>>n>>m>>k;
    //需要初始化，对角线为0
    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= n; j ++ ){
            if(i==j) d[i][j]=0;
            else d[i][j]=0x3f3f3f3f;
        }
    //输入
    while (m -- ){
        int a,b,c;
        scanf("%d%d%d", &a, &b,&c);
        //这里是因为可能存在重边
        d[a][b] = min(d[a][b], c);
    }
    floyd();
    //输出
    while (k -- ){
        int x,y;
        scanf("%d%d", &x, &y);
        //这里也要像之前得到INF/2
        //因为可能存在负权边减少INF
        if(d[x][y]>0x3f3f3f3f/2) puts("impossible"); 
        else printf("%d\n",d[x][y]);
    }
            
}






3.5 最小生成树

    最小生成树(MST: Minimum spanning tree)对应的图通常都是无向图；有两个最常见的算法：

  1. 普里姆算法Prim：类似Dijkstra；

        稠密图使用朴素版Prim：$O(n^2)$ ；

        稀疏图使用堆优化版：$O(m\log n)$，一般不常用；

  1. 克鲁斯卡尔Kruskal：$O(m\log m)$，用于稀疏图。

如果是稠密图，通常使用朴素Prim，如果是稀疏图通常使用Kruskal。



3.5.1 朴素版普里姆

    和Dijkstra算法类似，不同的地方时，Dj每次遍历找的是全局离源点最近的节点，但是Prim找的是离当前最小生成树最近的点。

    例题：给定一个 n 个点 m 条边的无向图，图中可能存在重边和自环，边权可能为负数。求最小生成树的树边权重之和，如果最小生成树不存在则输出 impossible。给定一张边带权的无向图 G=(V,E)，其中 V 表示图中点的集合，E 表示图中边的集合，n=|V|，m=|E|。由 V 中的全部 n 个顶点和 E 中 n−1 条边构成的无向连通子图被称为 G 的一棵生成树，其中边的权值之和最小的生成树被称为无向图 G 的最小生成树。 输入格式 第一行包含两个整数 n 和 m。

接下来 m 行，每行包含三个整数 u,v,w，表示点 u 和点 v 之间存在一条权值为 w 的边。

输出格式
共一行，若存在最小生成树，则输出一个整数，表示最小生成树的树边权重之和，如果最小生成树不存在则输出 impossible。

数据范围
1≤n≤500,
1≤m≤105,
图中涉及边的边权的绝对值均不超过 10000。

//dist中装的是节点距离联通块的距离，而不是距离源点距离
int g[N][N],st[N],dist[N],res;

int prime(){
    memset(dist, 0x3f, sizeof dist);
    //找到全局离联通块最近的节点
    for (int i = 0; i < n; i ++ ){
        int t=-1;
        for (int j = 1; j <= n; j ++ )
            //这里不能重复考虑已被统计的节点
            if(!st[j]&&(t==-1 || dist[t] > dist[j] ))
                t=j;
        //这里必须要检查i!=0，因为i==0时dist[t]==INF
        //因为还没有更新过dist数组
        if(i&&dist[t]==0x3f3f3f3f) return 0x3f3f3f3f;
        //这里必须先res+=dist[t]然后再更新dist数组
        //因为可能出现自环，当t==j，并且自环是最小的
        //会导致这里添加自环的值；而自环不可能出现在最小生成树中
        if(i) res+=dist[t];
        st[t]=1;

        //从最新统计的节点出发，更新其他所有节点
        for (int j = 1; j <= n; j ++ ){
            dist[j]=min(dist[j],g[t][j]);
        } 
    }
    return res;
}


3.5.2 Kruskal

    主要思想：把所有边按权重从小到大排序，枚举每条边，若a，b不在一个联通块，把a，b加到1个集合。

    例题：给定一个 n 个点 m 条边的无向图，图中可能存在重边和自环，边权可能为负数。求最小生成树的树边权重之和，如果最小生成树不存在则输出 impossible。给定一张边带权的无向图 G=(V,E)，其中 V 表示图中点的集合，E 表示图中边的集合，n=|V|，m=|E|。由 V 中的全部 n 个顶点和 E 中 n−1 条边构成的无向连通子图被称为 G 的一棵生成树，其中边的权值之和最小的生成树被称为无向图 G 的最小生成树。

int q[N];
//是对edge的遍历，所以使用结构体即可
struct edge{
    int a,b,c;
    bool operator <(edge &W)const{
        return c<W.c;
    }
}edges[M];
//并查集
int find(int a){
    if(q[a]!=a) q[a]=find(q[a]);
    return q[a];
}

int krs(){
    //排序边权，类似贪心
    sort(edges,edges+m);
    //初始化并查集
    for (int i = 1; i <= n; i ++ ) q[i]=i;
    
    //res为边权之和，cnt为遍历的节点数量
    int res=0,cnt=0;
    
    for (int i = 0; i < m; i ++ ){
        int a,b,c;
        //如果两个点没有在图中相连，这条边不会存在，也就不会被遍历
        a=edges[i].a,b=edges[i].b,c=edges[i].c;
        int u=find(a),v=find(b);
        //不是同一个根节点，也就是未加入同一个连通块
        if(u!=v){
            //修改的是跟节点的跟节点
            q[u]=v;
            res+=c;
            cnt++;
        }
    }
    //当两个联通块之间没有边，会导致遍历节点数量却一个
    //也就是不存在最小生成树
    if(cnt<n-1) return INF;
    return res;
    
}

注：并会导致环，因为对于跟节点的判断。







3.6 二分图

    题目通常涉及判断一个图是否是一个二分图，算法：

  1. 染色法: $O(m+n)$；

  2. 匈牙利算法: $O(mn)$，但是实际运行速度可能比$O(m+n)$ 好



3.6.1 染色法

    使用了图论中重要的性质：一个图是二分图，当切仅当一个图不存在奇数环（环的元素数量）。也就是说，假如给一个顶点的所有邻接点染色，当前顶点若为黑色，邻接点则为白色，反之亦然：一个二分图可以保证，在染色的过程中，不可能出现一个顶点被染成了黑色后又被染成了白色，或反之。

bool dfs(int x, int c){
    //给当前节点颜色
    color[x]=c;
    //遍历其所有邻接节点
    for (int i = h[x]; i != -1; i = ne[i] ){
        int t=e[i];
        //如果当前邻接节点没有染色
        if(!color[t]){
            //进行染色
            if(!dfs(t,3-c)) return false;
            //如果染了色，判断是否冲突
        }else if(color[t]==color[x]) return false;
    }
    return true;
}

int main()
{
    cin>>n>>m;
    memset(h, -1, sizeof h);
    while (m -- ){
        int a,b;
        scanf("%d%d", &a, &b);
        //邻接表
        add(a, b);
        add(b,a);
    }
    bool flag=true;
    //遍历每个节点，因为图中不一定只有一个连通块
    for (int i = 1; i <= n; i ++ ){
        if(!color[i]){
            if(!dfs(i,1)) {
                flag = false;
                break;
            }
        }
    }
    if(flag) cout<<"Yes";
    else cout<<"No";
}


3.6.2 匈牙利算法

    匈牙利算法可以判断而分数最大匹配数。最大匹配数的意思是：在一个二分图中，两变相连节点最大的一一对应的个数。

    例题：给定一个二分图，其中左半部包含 n1 个点（编号 1∼n1），右半部包含 n2 个点（编号 1∼n2），二分图共包含 m 条边。数据保证任意一条边的两个端点都不可能在同一部分中。请你求出二分图的最大匹配数。

二分图的匹配：给定一个二分图 G，在 G 的一个子图 M 中，M 的边集 {E} 中的任意两条边都不依附于同一个顶点，则称 M 是一个匹配。

二分图的最大匹配：所有匹配中包含边数最多的一组匹配被称为二分图的最大匹配，其边数即为最大匹配数。

输入格式
第一行包含三个整数 n1、 n2 和 m。

接下来 m 行，每行包含两个整数 u 和 v，表示左半部点集中的点 u 和右半部点集中的点 v 之间存在一条边。

输出格式
输出一个整数，表示二分图的最大匹配数。

//st用来标注当前右侧节点是否有匹配
bool st[N];
//match来标注当前节点匹配对象
int match[N];
int n1,n2,m;

void add(int a, int b){
    e[idx]=b, ne[idx]=h[a], h[a]=idx++;
}

bool find(int x){
    for (int i = h[x]; i != -1; i = ne[i] ){
        int j = e[i];
        //如果不存在st的检查，会导致
        //在递归的过程中，每次都会因为j而进入递归
        //出现死循环
        if(!st[j]){
            st[j]=true;
            if(match[j]==0 || find(match[j])){
                match[j]=x;
                return true;
            }
        }
    }
    return false;
}

int main()
{
    cin>>n1>>n2>>m;
    memset(h, -1, sizeof h);
    while (m -- ){
        int a,b;
        scanf("%d%d", &a, &b);
        add(a, b);
    }
    int res=0;
    for (int i = 1; i <= n1; i ++ ){
        //对于每一个新的左侧节点的匹配，都要清空st，
        //因为不能让之前的选择影响之后的选择以及递归时的
        //重新选择
        memset(st, false, sizeof st);
        if(find(i)) res++;
    }
    
    cout<<res;
}

注：在递归时，外部的st的设置不会影响递归内部的元素的选择，因为对于内部，这个对应的st也必然已经是true了。
