2.1 链表

    链表的实现有两种，第一种是通过结构体保存当前节点的数据和指向下一个/上一节点的指针，第二种是通过数组模拟链表，即用多个数组分别代表第 idx 个节点的数据，上下节点（静态链表）。



2.1.1 单链表

    单链表只保存下一个节点的“指针”。模拟方式：通过一个数组记录节点的值（e），一个数组记录下一个节点的指针(ne)。

如上图所示，它代表着一个由数组模拟的链表。Head可以理解为一个角标，它的值是第一个节点的idx，形式上是“指向”头节点。头节点的值为10，idx为0，它的下一个节点的idx为1。第二个节点则指向idx为5的元素，后者是尾节点。连续的节点的idx不一定连续，因为可能存在删除的操作。在数组模拟中，删除的节点实际上是直接被其上一个节点的ne跳过了，但实际上仍然存在于这个数组之中。虽然会造成空间的浪费，但是当节点数据特别多的时候，数组模拟创建速度比创建struct快得多，且后者能做的事情前者全能做。

需求：

  1. 在头部添加元素：给e[idx]赋值，让ne[idx]指向head所指向的节点（原头节点）并且让head等于idx，让它成为头节点。
  2. 在k节点的后面添加节点。这里的k是节点idx的值。类似添加头节点，只不过把head换成了某个节点
  3. 删除节点k后面的节点：实际上就是让k的ne[k]跳过下一个节点，即等于下一个节点的ne。
#include <iostream>
using namespace std;
#define N 100010
int m,head,idx;
int e[N],ne[N];

//初始化链表，用-1代表null，即尾部
void init(){
    head=-1;
    //idx从0开始分配，代表着共有几个节点被创建
    idx=0;
}

void remove (int k){
    //跳过下一个点
    ne[k]=ne[ne[k]];
}

void addH(int x){
    //这里可以合并成一行写，是用逗号分隔
    e[idx]=x, ne[idx]=head, head=idx++;
}

void add(int k, int x){
    e[idx]=x, ne[idx]=ne[k], ne[k]=idx++;
}

int main()
{
    char a;
    int b,c;
    init();

    //a是输入的指令
    cin>>a;
    if(a=='H'){
        cin>>b;
           addH(b);
    }
    if(a=='D'){
        cin>>b;
        if(!b) head=ne[head];
        else remove(b-1);
    }
    if(a=='I'){
        cin>>b>>c;
        add(b-1,c);
    }
    //遍历需要通过寻找下一个即诶单完成而不是i+1，因为idx不连续
for (int i = head; i != -1; i = ne[i]) cout << e[i] << ' ';
return 0;
}


2.1.2 双链表

    双链表即有指向左右节点的指针的链表，同样可以通过数组模拟。

//m：操作次数
int m,idx;
//这里的l和r分别代表一个节点做节点和右节点的idx
int e[N],l[N],r[N];

//在idx=a后面添加节点
void add(int a, int x){
    //赋值
    e[idx]=x;
    //搭建x
    l[idx]=a,r[idx]=r[a];
    //重构r[a]和a
    l[r[a]]=idx,r[a]=idx++;
}

void remove(int a){
    //把a跳过去
    l[r[a]]=l[a];
    r[l[a]]=r[a];
}

int main()
{
    //初始化
    //创建头节点和尾节点
    //这样相当于已经搭建了1个链表
    //解释为头节点的右边的节点是1，1是右端点
    l[1]=0, r[0]=1;
    idx=2;
    cin>>m;
    string op;
    //x：操作的值；b：操作的idx
    int x,b;
    while(m--){
        cin>>op;
        //首插
        if(op=="L"){
            cin>>x;
            add(0,x);
        }
        //尾插
        if(op=="R"){
            cin>>x;
            //即给右端点左边的节点后面插入节点
            add(l[1],x);
        }
        //给指定元素左边插入
        //b+1实际上是b+2-1，+2是因为最开始已经有两个节点了
        //-1是因为第n个插入的节点实际上idx是n-1
        if(op=="IL"){
            cin>>b>>x;
            add(l[b+1],x);
        }
        //给指定元素右边插入
        if(op=="IR"){
            cin>>b>>x;
            add(b+1,x);
        }
        //删除指定的节点
        if(op=="D"){
            cin>>b;
            remove(b+1);
        }
    }
    for (int i = r[0]; i != 1; i = r[i]) cout << e[i] << ' ';
}






2.2 栈和队列

    同样，本节介绍通过数组模拟栈和队列以及单调栈/单调队列。栈和队列的实现更加简单。



栈：LIFO：last in first out

//m：操作次数
int m;
//使用stk代表stack，tt代表栈顶
//默认栈顶从1开始
int stk[N], tt;

int main()
{
    cin >> m;
    while (m -- )
    {
        string op;
        int x;

        cin >> op;
        if (op == "push")
        {
            cin >> x;
            stk[ ++ tt] = x;
        }
        else if (op == "pop") tt -- ;
        else if (op == "empty") cout << (tt ? "NO" : "YES") << endl;
        //返回栈顶
        else cout << stk[tt] << endl;
    }
    return 0;
}

循环队列：循环利用使用数组最前面的部分

// hh 表示队头，tt表示队尾的后一个位置,从1开始

// 向队尾插入一个数
q[tt ++ ] = x;

if (tt == N) tt = 0;

hh ++ ;
//标志着一个循环

单调栈：本质上就是一个栈，栈内元素单调

例题：给定一个长度为 $N$ 的整数数列，输出每个数左边第一个比它小的数，如果不存在则输出 −1。

输入格式:

  第一行包含整数 N，表示数列长度。

  第二行包含 N 个整数，表示整数数列。

输出格式:

  共一行，包含 N 个整数，其中第 i 个数表示第 i 个数的左边第一个比它小的数，如果不存在则输出 −1。

数据范围:

  $1≤N≤10^5$   $1≤$ 数列中元素$≤10^9$

#define N 100010
int m;
int skt[N],tt;

int main()
{
    cin>>m;
    int x;
    for (int i = 0; i < m; i ++ ){
        scanf("%d", &x);
        //加入新输入的元素小于栈顶元素，那么该栈顶元素也将不会在未来的答案中出现
        //可以放心地把它删掉
        while(tt&&x<=skt[tt]) tt--;
        //如果到了栈底，说明前面所有数据都大于x
        if(!tt) {
            cout<<-1<<" ";
        }
        else cout<<skt[tt]<<" ";
        skt[++tt]=x;
    }
}
